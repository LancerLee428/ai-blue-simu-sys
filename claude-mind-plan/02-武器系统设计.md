# 武器系统设计方案

## 设计目标

构建一个专业的武器系统，支持导弹发射、飞行、制导、拦截和命中判定，为战术推演提供真实的火力打击能力。

## 核心架构

### 3. 武器系统核心

```typescript
// services/weapon-system.ts

export class WeaponSystem {
  private weapons = new Map<string, Weapon>();
  private onLaunch?: (event: WeaponLaunchEvent) => void;
  private onImpact?: (event: WeaponImpactEvent) => void;

  /**
   * 发射武器
   */
  launch(
    launcherId: string,
    targetId: string,
    weaponType: string,
    launchPosition: GeoPosition,
    launchTime: number,
  ): Weapon | null {
    const spec = WEAPON_DATABASE[weaponType];
    if (!spec) {
      console.warn(`未知武器类型: ${weaponType}`);
      return null;
    }

    const weapon: Weapon = {
      id: `weapon-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
      spec,
      launcherId,
      targetId,
      launchTime,
      launchPosition: { ...launchPosition },
      currentPosition: { ...launchPosition },
      velocity: { x: 0, y: 0, z: 0 },
      status: 'launching',
      trajectory: [{ ...launchPosition }],
      fuelRemaining: 1.0,
    };

    this.weapons.set(weapon.id, weapon);

    // 触发发射事件
    this.onLaunch?.({
      type: 'weapon-launch',
      time: launchTime,
      description: `${spec.name} 发射`,
      weaponId: weapon.id,
      launcherId,
      targetId,
      weaponType: spec.type,
    });

    return weapon;
  }

  /**
   * 更新所有武器状态
   */
  update(
    deltaTime: number,
    currentTime: number,
    entities: Map<string, GeoPosition>,
  ): void {
    for (const weapon of this.weapons.values()) {
      if (weapon.status === 'hit' || weapon.status === 'miss' ||
          weapon.status === 'intercepted' || weapon.status === 'fuel-out') {
        continue;
      }

      this.updateWeapon(weapon, deltaTime, currentTime, entities);
    }
  }

  /**
   * 更新单个武器
   */
  private updateWeapon(
    weapon: Weapon,
    deltaTime: number,
    currentTime: number,
    entities: Map<string, GeoPosition>,
  ): void {
    const elapsedTime = (currentTime - weapon.launchTime) / 1000; // 秒

    // 更新状态阶段
    if (elapsedTime < weapon.spec.launchDelay) {
      weapon.status = 'launching';
      return;
    } else if (elapsedTime < weapon.spec.launchDelay + 5) {
      weapon.status = 'boosting';
    } else if (this.getDistanceToTarget(weapon, entities) > 5000) {
      weapon.status = 'cruising';
    } else {
      weapon.status = 'terminal';
    }

    // 获取目标位置
    const targetPos = entities.get(weapon.targetId);
    if (!targetPos) {
      weapon.status = 'miss';
      return;
    }

    // 计算制导指令
    const guidance = this.calculateGuidance(weapon, targetPos);

    // 更新速度和位置
    this.updateVelocity(weapon, guidance, deltaTime);
    this.updatePosition(weapon, deltaTime);

    // 燃料消耗
    weapon.fuelRemaining -= deltaTime / (weapon.spec.maxRange / (weapon.spec.maxSpeed * 340));
    if (weapon.fuelRemaining <= 0) {
      weapon.status = 'fuel-out';
      return;
    }

    // 检查命中
    const distanceToTarget = this.calculateDistance(weapon.currentPosition, targetPos);
    if (distanceToTarget < weapon.spec.killRadius) {
      weapon.status = 'hit';
      this.onImpact?.({
        type: 'weapon-impact',
        time: currentTime,
        description: `${weapon.spec.name} 命中目标`,
        weaponId: weapon.id,
        targetId: weapon.targetId,
        hitPosition: { ...weapon.currentPosition },
        damage: this.calculateDamage(weapon, distanceToTarget),
      });
    }

    // 记录轨迹
    weapon.trajectory.push({ ...weapon.currentPosition });
  }

  /**
   * 计算制导指令 (比例导引法)
   */
  private calculateGuidance(weapon: Weapon, targetPos: GeoPosition): Vector3D {
    const dx = targetPos.longitude - weapon.currentPosition.longitude;
    const dy = targetPos.latitude - weapon.currentPosition.latitude;
    const dz = targetPos.altitude - weapon.currentPosition.altitude;

    const distance = Math.sqrt(dx * dx + dy * dy + dz * dz);

    // 归一化方向
    return {
      x: dx / distance,
      y: dy / distance,
      z: dz / distance,
    };
  }

  /**
   * 更新速度矢量
   */
  private updateVelocity(weapon: Weapon, guidance: Vector3D, deltaTime: number): void {
    const maxSpeed = weapon.spec.maxSpeed * 340; // 马赫转 m/s
    const acceleration = weapon.spec.acceleration;

    // 根据制导指令调整速度方向
    weapon.velocity.x += guidance.x * acceleration * deltaTime;
    weapon.velocity.y += guidance.y * acceleration * deltaTime;
    weapon.velocity.z += guidance.z * acceleration * deltaTime;

    // 限制最大速度
    const speed = Math.sqrt(
      weapon.velocity.x ** 2 +
      weapon.velocity.y ** 2 +
      weapon.velocity.z ** 2
    );
    if (speed > maxSpeed) {
      const scale = maxSpeed / speed;
      weapon.velocity.x *= scale;
      weapon.velocity.y *= scale;
      weapon.velocity.z *= scale;
    }
  }

  /**
   * 更新位置
   */
  private updatePosition(weapon: Weapon, deltaTime: number): void {
    // 简化：直接用速度更新位置（实际应考虑地球曲率）
    weapon.currentPosition.longitude += weapon.velocity.x * deltaTime / 111320;
    weapon.currentPosition.latitude += weapon.velocity.y * deltaTime / 110540;
    weapon.currentPosition.altitude += weapon.velocity.z * deltaTime;
  }

  /**
   * 计算损伤
   */
  private calculateDamage(weapon: Weapon, distance: number): number {
    // 损伤随距离衰减
    const ratio = 1 - (distance / weapon.spec.killRadius);
    return weapon.spec.warheadWeight * Math.max(0, ratio);
  }

  private getDistanceToTarget(weapon: Weapon, entities: Map<string, GeoPosition>): number {
    const targetPos = entities.get(weapon.targetId);
    if (!targetPos) return Infinity;
    return this.calculateDistance(weapon.currentPosition, targetPos);
  }

  private calculateDistance(pos1: GeoPosition, pos2: GeoPosition): number {
    const R = 6371000;
    const lat1 = (pos1.latitude * Math.PI) / 180;
    const lat2 = (pos2.latitude * Math.PI) / 180;
    const dLat = lat2 - lat1;
    const dLon = ((pos2.longitude - pos1.longitude) * Math.PI) / 180;

    const a =
      Math.sin(dLat / 2) * Math.sin(dLat / 2) +
      Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLon / 2) * Math.sin(dLon / 2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

    const horizontalDist = R * c;
    const verticalDist = pos2.altitude - pos1.altitude;

    return Math.sqrt(horizontalDist ** 2 + verticalDist ** 2);
  }

  setOnLaunch(callback: (event: WeaponLaunchEvent) => void): void {
    this.onLaunch = callback;
  }

  setOnImpact(callback: (event: WeaponImpactEvent) => void): void {
    this.onImpact = callback;
  }

  getWeapons(): Map<string, Weapon> {
    return this.weapons;
  }

  reset(): void {
    this.weapons.clear();
  }
}
```

      this.explosionRenderer.createExplosion(event.hitPosition, event.damage);
    });
  }

  private runAnimationLoop(): void {
    // ... 现有代码

    // 更新武器
    this.weaponSystem.update(
      realDeltaMs / 1000,
      this.virtualElapsedMs,
      this.entityPositions,
    );

    // 渲染武器
    this.renderer.renderWeapons(this.weaponSystem.getWeapons());
  }
}
