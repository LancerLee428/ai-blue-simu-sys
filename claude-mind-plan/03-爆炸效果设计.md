# 爆炸效果设计方案

## 设计目标

实现专业的爆炸可视化效果，包括火球、冲击波、烟雾、碎片等，为武器命中提供真实的视觉反馈。

## 核心架构

### 1. 类型系统扩展

```typescript
// types/tactical-scenario.ts 新增

/**
 * 爆炸类型
 */
export type ExplosionType =
  | 'missile-impact'    // 导弹命中
  | 'bomb-impact'       // 炸弹命中
  | 'ship-explosion'    // 舰艇爆炸
  | 'aircraft-crash'    // 飞机坠毁
  | 'fuel-explosion'    // 燃料爆炸
  | 'ammunition'        // 弹药殉爆
  | 'nuclear';          // 核爆炸

/**
 * 爆炸实例
 */
export interface Explosion {
  id: string;
  type: ExplosionType;
  position: GeoPosition;
  startTime: number;      // 爆炸开始时间 (虚拟时间 ms)
  duration: number;       // 持续时间 (ms)
  radius: number;         // 爆炸半径 (米)
  intensity: number;      // 强度 (0-1)
  damage: number;         // 伤害值
  status: 'active' | 'fading' | 'complete';
}

/**
 * 爆炸事件
 */
export interface ExplosionEvent extends TacticalEvent {
  type: 'explosion';
  explosionId: string;
  explosionType: ExplosionType;
  position: GeoPosition;
  radius: number;
  affectedEntities: string[]; // 受影响的实体 ID
}
```

### 2. 爆炸渲染器

```typescript
// services/explosion-renderer.ts

import * as Cesium from 'cesium';
import type { Explosion, ExplosionType, GeoPosition } from '../types/tactical-scenario';

/**
 * 爆炸渲染器
 * 使用 Cesium 粒子系统和几何体渲染爆炸效果
 */
export class ExplosionRenderer {
  private viewer: Cesium.Viewer;
  private explosions = new Map<string, Explosion>();
  private particleSystems = new Map<string, Cesium.ParticleSystem>();

  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer;
  }

  /**
   * 创建爆炸
   */
  createExplosion(
    position: GeoPosition,
    damage: number,
    type: ExplosionType = 'missile-impact',
    startTime: number,
  ): Explosion {
    const explosion: Explosion = {
      id: `explosion-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`,
      type,
      position: { ...position },
      startTime,
      duration: this.getExplosionDuration(type),
      radius: this.calculateExplosionRadius(damage, type),
      intensity: Math.min(1.0, damage / 500),
      damage,
      status: 'active',
    };

    this.explosions.set(explosion.id, explosion);
    this.renderExplosion(explosion);

    return explosion;
  }

  /**
   * 渲染爆炸效果
   */
  private renderExplosion(explosion: Explosion): void {
    const cartesian = Cesium.Cartesian3.fromDegrees(
      explosion.position.longitude,
      explosion.position.latitude,
      explosion.position.altitude,
    );

    // 1. 火球（Billboard + 动画缩放）
    this.renderFireball(explosion, cartesian);

    // 2. 冲击波（扩散圆环）
    this.renderShockwave(explosion, cartesian);

    // 3. 烟雾柱（粒子系统）
    this.renderSmokePlume(explosion, cartesian);

    // 4. 碎片（粒子系统）
    if (explosion.type !== 'nuclear') {
      this.renderDebris(explosion, cartesian);
    }

    // 5. 闪光（Point + 衰减）
    this.renderFlash(explosion, cartesian);
  }

  /**
   * 渲染火球
   */
  private renderFireball(explosion: Explosion, position: Cesium.Cartesian3): void {
    const duration = explosion.duration;
    const maxScale = explosion.radius / 10;

    // 使用 Billboard 渲染火球纹理
    const billboard = this.viewer.entities.add({
      id: `${explosion.id}-fireball`,
      position,
      billboard: {
        image: this.getFireballTexture(explosion.type),
        scale: 0.1,
        color: Cesium.Color.WHITE,
        heightReference: Cesium.HeightReference.NONE,
      },
    });

    // 动画：缩放 + 淡出
    const startTime = Cesium.JulianDate.now();
    const stopTime = Cesium.JulianDate.addSeconds(startTime, duration / 1000, new Cesium.JulianDate());

    const scaleProperty = new Cesium.SampledProperty(Number);
    scaleProperty.addSample(startTime, 0.1);
    scaleProperty.addSample(
      Cesium.JulianDate.addSeconds(startTime, duration / 2000, new Cesium.JulianDate()),
      maxScale,
    );
    scaleProperty.addSample(stopTime, maxScale * 1.5);

    const colorProperty = new Cesium.SampledProperty(Cesium.Color);
    colorProperty.addSample(startTime, Cesium.Color.WHITE);
    colorProperty.addSample(
      Cesium.JulianDate.addSeconds(startTime, duration / 3000, new Cesium.JulianDate()),
      Cesium.Color.ORANGE,
    );
    colorProperty.addSample(stopTime, Cesium.Color.BLACK.withAlpha(0));

    billboard.billboard!.scale = scaleProperty as any;
    billboard.billboard!.color = colorProperty as any;

    // 定时清理
    setTimeout(() => {
      this.viewer.entities.remove(billboard);
    }, duration);
  }

  /**
   * 渲染冲击波
   */
  private renderShockwave(explosion: Explosion, position: Cesium.Cartesian3): void {
    const duration = explosion.duration * 0.3; // 冲击波持续时间较短
    const maxRadius = explosion.radius;

    // 使用 EllipseGraphics 渲染扩散圆环
    const shockwave = this.viewer.entities.add({
      id: `${explosion.id}-shockwave`,
      position,
      ellipse: {
        semiMajorAxis: 10,
        semiMinorAxis: 10,
        material: Cesium.Color.WHITE.withAlpha(0.8),
        outline: true,
        outlineColor: Cesium.Color.ORANGE.withAlpha(0.9),
        outlineWidth: 3,
        height: explosion.position.altitude,
        heightReference: Cesium.HeightReference.NONE,
      },
    });

    // 动画：半径扩大 + 透明度降低
    const startTime = Cesium.JulianDate.now();
    const stopTime = Cesium.JulianDate.addSeconds(startTime, duration / 1000, new Cesium.JulianDate());

    const radiusProperty = new Cesium.SampledProperty(Number);
    radiusProperty.addSample(startTime, 10);
    radiusProperty.addSample(stopTime, maxRadius);

    const materialProperty = new Cesium.ColorMaterialProperty(
      new Cesium.SampledProperty(Cesium.Color),
    );
    (materialProperty.color as Cesium.SampledProperty).addSample(
      startTime,
      Cesium.Color.WHITE.withAlpha(0.8),
    );
    (materialProperty.color as Cesium.SampledProperty).addSample(
      stopTime,
      Cesium.Color.WHITE.withAlpha(0),
    );

    shockwave.ellipse!.semiMajorAxis = radiusProperty as any;
    shockwave.ellipse!.semiMinorAxis = radiusProperty as any;
    shockwave.ellipse!.material = materialProperty as any;

    // 定时清理
    setTimeout(() => {
      this.viewer.entities.remove(shockwave);
    }, duration);
  }

  /**
   * 渲染烟雾柱（粒子系统）
   */
  private renderSmokePlume(explosion: Explosion, position: Cesium.Cartesian3): void {
    const particleSystem = this.viewer.scene.primitives.add(
      new Cesium.ParticleSystem({
        image: this.getSmokeTexture(),
        startColor: Cesium.Color.GRAY.withAlpha(0.7),
        endColor: Cesium.Color.DARKGRAY.withAlpha(0.0),
        startScale: explosion.radius / 50,
        endScale: explosion.radius / 20,
        minimumParticleLife: 2.0,
        maximumParticleLife: 4.0,
        minimumSpeed: 5.0,
        maximumSpeed: 15.0,
        imageSize: new Cesium.Cartesian2(25, 25),
        emissionRate: 50.0,
        lifetime: explosion.duration / 1000,
        emitter: new Cesium.CircleEmitter(explosion.radius / 10),
        modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
      }),
    );

    this.particleSystems.set(explosion.id, particleSystem);

    // 定时清理
    setTimeout(() => {
      this.viewer.scene.primitives.remove(particleSystem);
      this.particleSystems.delete(explosion.id);
    }, explosion.duration);
  }

  /**
   * 渲染碎片（粒子系统）
   */
  private renderDebris(explosion: Explosion, position: Cesium.Cartesian3): void {
    const particleSystem = this.viewer.scene.primitives.add(
      new Cesium.ParticleSystem({
        image: this.getDebrisTexture(),
        startColor: Cesium.Color.ORANGE.withAlpha(1.0),
        endColor: Cesium.Color.BLACK.withAlpha(0.0),
        startScale: 2.0,
        endScale: 0.5,
        minimumParticleLife: 0.5,
        maximumParticleLife: 2.0,
        minimumSpeed: 20.0,
        maximumSpeed: 50.0,
        imageSize: new Cesium.Cartesian2(5, 5),
        emissionRate: 200.0,
        lifetime: 0.5, // 碎片发射持续时间短
        bursts: [
          new Cesium.ParticleBurst({
            time: 0.0,
            minimum: 100,
            maximum: 200,
          }),
        ],
        emitter: new Cesium.SphereEmitter(explosion.radius / 20),
        modelMatrix: Cesium.Transforms.eastNorthUpToFixedFrame(position),
      }),
    );

    // 定时清理
    setTimeout(() => {
      this.viewer.scene.primitives.remove(particleSystem);
    }, 2500);
  }

  /**
   * 渲染闪光
   */
  private renderFlash(explosion: Explosion, position: Cesium.Cartesian3): void {
    const flash = this.viewer.entities.add({
      id: `${explosion.id}-flash`,
      position,
      point: {
        pixelSize: 50,
        color: Cesium.Color.WHITE,
        outlineColor: Cesium.Color.ORANGE,
        outlineWidth: 3,
      },
    });

    // 快速淡出
    const startTime = Cesium.JulianDate.now();
    const stopTime = Cesium.JulianDate.addSeconds(startTime, 0.2, new Cesium.JulianDate());

    const colorProperty = new Cesium.SampledProperty(Cesium.Color);
    colorProperty.addSample(startTime, Cesium.Color.WHITE);
    colorProperty.addSample(stopTime, Cesium.Color.WHITE.withAlpha(0));

    flash.point!.color = colorProperty as any;

    // 定时清理
    setTimeout(() => {
      this.viewer.entities.remove(flash);
    }, 200);
  }

  /**
   * 更新爆炸状态
   */
  update(currentTime: number): void {
    for (const explosion of this.explosions.values()) {
      const elapsed = currentTime - explosion.startTime;

      if (elapsed > explosion.duration) {
        explosion.status = 'complete';
        this.explosions.delete(explosion.id);
      } else if (elapsed > explosion.duration * 0.7) {
        explosion.status = 'fading';
      }
    }
  }

  /**
   * 计算爆炸半径
   */
  private calculateExplosionRadius(damage: number, type: ExplosionType): number {
    const baseRadius = Math.sqrt(damage) * 2;

    switch (type) {
      case 'nuclear':
        return baseRadius * 100;
      case 'ship-explosion':
        return baseRadius * 5;
      case 'fuel-explosion':
        return baseRadius * 3;
      default:
        return baseRadius;
    }
  }

  /**
   * 获取爆炸持续时间
   */
  private getExplosionDuration(type: ExplosionType): number {
    switch (type) {
      case 'nuclear':
        return 30000; // 30秒
      case 'ship-explosion':
        return 8000;
      case 'fuel-explosion':
        return 6000;
      default:
        return 3000; // 3秒
    }
  }

  /**
   * 获取火球纹理
   */
  private getFireballTexture(type: ExplosionType): string {
    // 实际应用中应使用真实纹理图片
    // 这里返回 Canvas 生成的径向渐变
    const canvas = document.createElement('canvas');
    canvas.width = 256;
    canvas.height = 256;
    const ctx = canvas.getContext('2d')!;

    const gradient = ctx.createRadialGradient(128, 128, 0, 128, 128, 128);
    gradient.addColorStop(0, 'rgba(255, 255, 200, 1)');
    gradient.addColorStop(0.3, 'rgba(255, 150, 0, 1)');
    gradient.addColorStop(0.6, 'rgba(255, 50, 0, 0.8)');
    gradient.addColorStop(1, 'rgba(50, 0, 0, 0)');

    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, 256, 256);

    return canvas.toDataURL();
  }

  /**
   * 获取烟雾纹理
   */
  private getSmokeTexture(): string {
    const canvas = document.createElement('canvas');
    canvas.width = 128;
    canvas.height = 128;
    const ctx = canvas.getContext('2d')!;

    const gradient = ctx.createRadialGradient(64, 64, 0, 64, 64, 64);
    gradient.addColorStop(0, 'rgba(100, 100, 100, 0.8)');
    gradient.addColorStop(0.5, 'rgba(80, 80, 80, 0.5)');
    gradient.addColorStop(1, 'rgba(60, 60, 60, 0)');

    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, 128, 128);

    return canvas.toDataURL();
  }

  /**
   * 获取碎片纹理
   */
  private getDebrisTexture(): string {
    const canvas = document.createElement('canvas');
    canvas.width = 16;
    canvas.height = 16;
    const ctx = canvas.getContext('2d')!;

    ctx.fillStyle = 'rgba(200, 100, 0, 1)';
    ctx.fillRect(0, 0, 16, 16);

    return canvas.toDataURL();
  }

  reset(): void {
    this.explosions.clear();
    for (const ps of this.particleSystems.values()) {
      this.viewer.scene.primitives.remove(ps);
    }
    this.particleSystems.clear();
  }
}
```

        affectedEntities: [event.targetId],
      });
    });
  }

  }
}

## 性能优化

### 1. 粒子系统池化
```typescript
class ParticleSystemPool {
  private pool: Cesium.ParticleSystem[] = [];

  acquire(config: any): Cesium.ParticleSystem {
    return this.pool.pop() || new Cesium.ParticleSystem(config);
  }

  release(ps: Cesium.ParticleSystem): void {
    this.pool.push(ps);
  }
}
```

private readonly MAX_ACTIVE_EXPLOSIONS = 10;

createExplosion(...): Explosion | null {
  if (this.explosions.size >= this.MAX_ACTIVE_EXPLOSIONS) {
    // 移除最早的爆炸
    const oldest = Array.from(this.explosions.values())[0];
    this.removeExplosion(oldest.id);
