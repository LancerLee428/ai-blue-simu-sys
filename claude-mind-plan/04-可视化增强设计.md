# 可视化增强设计方案

## 设计目标

在现有基础上增强态势可视化效果，包括雷达扫描动画、导弹尾焰、电子战波纹、传感器光束等专业效果。

## 1. 雷达扫描动画

### 设计思路
在现有 `RadarBeamRenderer` 基础上增加旋转扫描效果，模拟真实雷达工作状态。

### 实现方案

```typescript
// services/sensor-animation.ts

import * as Cesium from 'cesium';
import type { DetectionZone, GeoPosition } from '../types/tactical-scenario';
import { FORCE_COLORS } from './cesium-graphics';

/**
 * 传感器动画渲染器
 * 负责雷达扫描、光束旋转等动态效果
 */
export class SensorAnimationRenderer {
  private viewer: Cesium.Viewer;
  private scanningBeams = new Map<string, ScanningBeam>();
  private animationFrameId: number | null = null;

  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer;
  }

  /**
   * 启动雷达扫描动画
   */
  startRadarScan(
    zone: DetectionZone,
    entityHeading: number,
    scanSpeed: number = 60, // 度/秒
  ): void {
    const beam: ScanningBeam = {
      id: `scan-${zone.entityId}`,
      zoneId: zone.entityId,
      position: zone.center,
      currentAngle: entityHeading,
      scanSpeed,
      radius: zone.radiusMeters,
      side: zone.side,
      active: true,
    };

    this.scanningBeams.set(beam.id, beam);

    // 启动动画循环
    if (this.animationFrameId === null) {
      this.runScanAnimation();
    }
  }

  /**
   * 停止雷达扫描
   */
  stopRadarScan(entityId: string): void {
    const beamId = `scan-${entityId}`;
    const beam = this.scanningBeams.get(beamId);
    if (beam) {
      beam.active = false;
      this.scanningBeams.delete(beamId);
      this.viewer.entities.removeById(beamId);
    }

    if (this.scanningBeams.size === 0 && this.animationFrameId !== null) {
      cancelAnimationFrame(this.animationFrameId);
      this.animationFrameId = null;
    }
  }

  /**
   * 扫描动画循环
   */
  private runScanAnimation = (): void => {
    const now = performance.now();
    const deltaTime = this.lastUpdateTime ? (now - this.lastUpdateTime) / 1000 : 0;
    this.lastUpdateTime = now;

    for (const beam of this.scanningBeams.values()) {
      if (!beam.active) continue;

      // 更新扫描角度
      beam.currentAngle = (beam.currentAngle + beam.scanSpeed * deltaTime) % 360;

      // 渲染扫描波束
      this.renderScanBeam(beam);
    }

    this.animationFrameId = requestAnimationFrame(this.runScanAnimation);
  };

  private lastUpdateTime: number = 0;

  /**
   * 渲染单个扫描波束
   */
  private renderScanBeam(beam: ScanningBeam): void {
    const colors = FORCE_COLORS[beam.side];
    const beamWidth = 15; // 波束宽度（度）

    // 移除旧的波束实体
    this.viewer.entities.removeById(beam.id);

    // 计算扇形顶点
    const center = Cesium.Cartesian3.fromDegrees(
      beam.position.longitude,
      beam.position.latitude,
      beam.position.altitude,
    );

    const positions = [center];
    const startAngle = Cesium.Math.toRadians(beam.currentAngle - beamWidth / 2);
    const endAngle = Cesium.Math.toRadians(beam.currentAngle + beamWidth / 2);

    // 生成扇形边界
    const segments = 20;
    for (let i = 0; i <= segments; i++) {
      const angle = startAngle + ((endAngle - startAngle) * i) / segments;
      const offsetLon =
        (beam.radius / 111320) * Math.sin(angle) / Math.cos(beam.position.latitude * Math.PI / 180);
      const offsetLat = (beam.radius / 110540) * Math.cos(angle);
      positions.push(
        Cesium.Cartesian3.fromDegrees(
          beam.position.longitude + offsetLon,
          beam.position.latitude + offsetLat,
          beam.position.altitude,
        ),
      );
    }
    positions.push(center);

    // 创建扫描波束实体
    this.viewer.entities.add({
      id: beam.id,
      polygon: {
        hierarchy: new Cesium.PolygonHierarchy(positions),
        material: colors.detectionFill.withAlpha(0.3),
        outline: true,
        outlineColor: colors.detectionOutline.withAlpha(0.8),
        outlineWidth: 2,
        height: beam.position.altitude,
        heightReference:
          beam.position.altitude < 10
            ? Cesium.HeightReference.CLAMP_TO_GROUND
            : Cesium.HeightReference.NONE,
      },
    });
  }

  reset(): void {
    for (const beam of this.scanningBeams.values()) {
      this.viewer.entities.removeById(beam.id);
    }
    this.scanningBeams.clear();

    if (this.animationFrameId !== null) {
      cancelAnimationFrame(this.animationFrameId);
      this.animationFrameId = null;
    }
  }
}

interface ScanningBeam {
  id: string;
  zoneId: string;
  position: GeoPosition;
  currentAngle: number;
  scanSpeed: number;
  radius: number;
  side: 'red' | 'blue';
  active: boolean;
}
```

## 2. 导弹尾焰效果

### 实现方案

```typescript
// services/weapon-trail-renderer.ts

import * as Cesium from 'cesium';
import type { Weapon } from '../types/tactical-scenario';

/**
 * 武器尾焰渲染器
 */
export class WeaponTrailRenderer {
  private viewer: Cesium.Viewer;
  private trails = new Map<string, Cesium.Entity>();

  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer;
  }

  /**
   * 为武器添加尾焰
   */
  addTrail(weapon: Weapon): void {
    // 根据武器状态决定尾焰颜色和强度
    const trailColor = this.getTrailColor(weapon);
    const trailWidth = this.getTrailWidth(weapon);

    const trail = this.viewer.entities.add({
      id: `trail-${weapon.id}`,
      polyline: {
        positions: new Cesium.CallbackProperty(() => {
          // 动态获取最近 N 个轨迹点
          const recentPoints = weapon.trajectory.slice(-20);
          return recentPoints.map((pos) =>
            Cesium.Cartesian3.fromDegrees(pos.longitude, pos.latitude, pos.altitude),
          );
        }, false),
        width: trailWidth,
        material: new Cesium.PolylineGlowMaterialProperty({
          glowPower: 0.3,
          color: trailColor,
        }),
      },
    });

    this.trails.set(weapon.id, trail);
  }

  /**
   * 移除尾焰
   */
  removeTrail(weaponId: string): void {
    const trail = this.trails.get(weaponId);
    if (trail) {
      this.viewer.entities.remove(trail);
      this.trails.delete(weaponId);
    }
  }

  /**
   * 获取尾焰颜色
   */
  private getTrailColor(weapon: Weapon): Cesium.Color {
    switch (weapon.status) {
      case 'boosting':
        return Cesium.Color.ORANGE.withAlpha(0.9);
      case 'cruising':
        return Cesium.Color.YELLOW.withAlpha(0.7);
      case 'terminal':
        return Cesium.Color.RED.withAlpha(0.8);
      default:
        return Cesium.Color.WHITE.withAlpha(0.5);
    }
  }

  /**
   * 获取尾焰宽度
   */
  private getTrailWidth(weapon: Weapon): number {
    const baseWidth = 3;
    const speedFactor = weapon.spec.maxSpeed / 2; // 速度越快，尾焰越粗
    return baseWidth * speedFactor;
  }

  reset(): void {
    for (const trail of this.trails.values()) {
      this.viewer.entities.remove(trail);
    }
    this.trails.clear();
  }
}
```

## 3. 电子战干扰波纹

### 实现方案

```typescript
// services/ew-visualization.ts

import * as Cesium from 'cesium';
import type { GeoPosition } from '../types/tactical-scenario';

/**
 * 电子战可视化渲染器
 */
export class EWVisualizationRenderer {
  private viewer: Cesium.Viewer;
  private jammerPulses = new Map<string, JammerPulse>();
  private animationFrameId: number | null = null;

  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer;
  }

  /**
   * 启动干扰脉冲动画
   */
  startJammerPulse(
    entityId: string,
    position: GeoPosition,
    radius: number,
  ): void {
    const pulse: JammerPulse = {
      id: `jammer-${entityId}`,
      position,
      maxRadius: radius,
      currentRadius: 0,
      pulseSpeed: radius / 2, // 2秒完成一次脉冲
      active: true,
    };

    this.jammerPulses.set(pulse.id, pulse);

    if (this.animationFrameId === null) {
      this.runPulseAnimation();
    }
  }

  /**
   * 停止干扰脉冲
   */
  stopJammerPulse(entityId: string): void {
    const pulseId = `jammer-${entityId}`;
    const pulse = this.jammerPulses.get(pulseId);
    if (pulse) {
      pulse.active = false;
      this.jammerPulses.delete(pulseId);
      this.viewer.entities.removeById(pulseId);
    }

    if (this.jammerPulses.size === 0 && this.animationFrameId !== null) {
      cancelAnimationFrame(this.animationFrameId);
      this.animationFrameId = null;
    }
  }

  /**
   * 脉冲动画循环
   */
  private runPulseAnimation = (): void => {
    const now = performance.now();
    const deltaTime = this.lastUpdateTime ? (now - this.lastUpdateTime) / 1000 : 0;
    this.lastUpdateTime = now;

    for (const pulse of this.jammerPulses.values()) {
      if (!pulse.active) continue;

      // 更新脉冲半径
      pulse.currentRadius += pulse.pulseSpeed * deltaTime;

      // 重置脉冲
      if (pulse.currentRadius >= pulse.maxRadius) {
        pulse.currentRadius = 0;
      }

      // 渲染脉冲
      this.renderPulse(pulse);
    }

    this.animationFrameId = requestAnimationFrame(this.runPulseAnimation);
  };

  private lastUpdateTime: number = 0;

  /**
   * 渲染单个脉冲
   */
  private renderPulse(pulse: JammerPulse): void {
    // 移除旧的脉冲实体
    this.viewer.entities.removeById(pulse.id);

    // 计算透明度（从中心向外衰减）
    const alpha = 1 - pulse.currentRadius / pulse.maxRadius;

    // 创建脉冲圆环
    this.viewer.entities.add({
      id: pulse.id,
      position: Cesium.Cartesian3.fromDegrees(
        pulse.position.longitude,
        pulse.position.latitude,
        pulse.position.altitude,
      ),
      ellipse: {
        semiMajorAxis: pulse.currentRadius,
        semiMinorAxis: pulse.currentRadius,
        material: Cesium.Color.ORANGE.withAlpha(alpha * 0.3),
        outline: true,
        outlineColor: Cesium.Color.ORANGE.withAlpha(alpha * 0.8),
        outlineWidth: 2,
        height: pulse.position.altitude,
        heightReference:
          pulse.position.altitude < 10
            ? Cesium.HeightReference.CLAMP_TO_GROUND
            : Cesium.HeightReference.NONE,
      },
    });
  }

  reset(): void {
    for (const pulse of this.jammerPulses.values()) {
      this.viewer.entities.removeById(pulse.id);
    }
    this.jammerPulses.clear();

    if (this.animationFrameId !== null) {
      cancelAnimationFrame(this.animationFrameId);
      this.animationFrameId = null;
    }
  }
}

interface JammerPulse {
  id: string;
  position: GeoPosition;
  maxRadius: number;
  currentRadius: number;
  pulseSpeed: number;
  active: boolean;
}
```

## 4. 集成到渲染器

### MapRenderer 扩展

```typescript
// services/map-renderer.ts 扩展

export class MapRenderer {
  private sensorAnimation: SensorAnimationRenderer;
  private weaponTrailRenderer: WeaponTrailRenderer;
  private ewVisualization: EWVisualizationRenderer;

  constructor(viewer: Cesium.Viewer) {
    this.viewer = viewer;
    this.sensorAnimation = new SensorAnimationRenderer(viewer);
    this.weaponTrailRenderer = new WeaponTrailRenderer(viewer);
    this.ewVisualization = new EWVisualizationRenderer(viewer);
  }

  /**
   * 渲染探测范围（增强版）
   */
  renderDetectionZones(zones: DetectionZone[], enableAnimation: boolean = true): void {
    zones.forEach((zone) => {
      // 渲染静态探测范围
      this.renderDetectionZone(zone);

      // 启动雷达扫描动画
      if (enableAnimation && this.isRadarZone(zone)) {
        const entity = this.findEntity(zone.entityId);
        if (entity) {
          this.sensorAnimation.startRadarScan(zone, entity.position.heading || 0);
        }
      }
    });
  }

  /**
   * 渲染武器（增强版）
   */
  renderWeapons(weapons: Map<string, Weapon>): void {
    for (const weapon of weapons.values()) {
      // 渲染武器实体
      this.renderWeaponEntity(weapon);

      // 添加尾焰
      if (weapon.status === 'boosting' || weapon.status === 'cruising' || weapon.status === 'terminal') {
        if (!this.weaponTrailRenderer.hasTrail(weapon.id)) {
          this.weaponTrailRenderer.addTrail(weapon);
        }
      } else {
        this.weaponTrailRenderer.removeTrail(weapon.id);
      }
    }
  }

  /**
   * 渲染电子战效果
   */
  renderEWEffects(jammers: Map<string, JammerZone>): void {
    for (const jammer of jammers.values()) {
      if (jammer.active) {
        this.ewVisualization.startJammerPulse(
          jammer.entityId,
          jammer.position,
          jammer.radius,
        );
      } else {
        this.ewVisualization.stopJammerPulse(jammer.entityId);
      }
    }
  }

  private isRadarZone(zone: DetectionZone): boolean {
    return zone.label?.includes('雷达') || false;
  }
}
```
